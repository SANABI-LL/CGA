02

Lambda handler

File: backend/lambdas/query/handler.ts. Deploy with AWS SAM or CDK. Set env var GEODATA_BUCKET=campusgeo-geodata-491117467175.

// backend/lambdas/query/handler.ts
import { BedrockRuntimeClient, ConverseCommand } from '@aws-sdk/client-bedrock-runtime';
import { S3Client, GetObjectCommand } from '@aws-sdk/client-s3';
import * as turf from '@turf/turf';
import * as fs from 'fs';

const bedrock = new BedrockRuntimeClient({ region: 'us-east-1' });
const s3 = new S3Client({ region: 'us-east-1' });
const BUCKET = process.env.GEODATA_BUCKET ?? 'campusgeo-geodata-491117467175';
const MODEL = 'anthropic.claude-3-5-sonnet-20241022-v2:0';

// ── GeoJSON cache (warm Lambda reuses /tmp) ──────────────────────
const memCache: Record<string, any> = {};
async function loadLayer(key: string): Promise<any> {
  if (memCache[key]) return memCache[key];
  const tmp = `/tmp/${key.replace(/\//g, '_')}`;
  if (fs.existsSync(tmp)) { memCache[key] = JSON.parse(fs.readFileSync(tmp, 'utf8')); return memCache[key]; }
  const res = await s3.send(new GetObjectCommand({ Bucket: BUCKET, Key: key }));
  const body = await res.Body!.transformToString();
  fs.writeFileSync(tmp, body); memCache[key] = JSON.parse(body);
  return memCache[key];
}

// ── Tool definitions (from Intent Taxonomy spec) ──────────────────
const TOOLS = [
  { toolSpec: { name: 'locate_feature', description: 'Find one named building or place and center the map on it.',
    inputSchema: { json: { type: 'object', required: ['name'], properties: {
      name: { type: 'string', description: 'Building name or known alias' },
      layer: { type: 'string', enum: ['buildings', 'trees'] } }}}}},

  { toolSpec: { name: 'filter_features', description: 'Return features matching attribute predicates (year, architect, ownership, species, condition).',
    inputSchema: { json: { type: 'object', required: ['layer', 'filters'], properties: {
      layer: { type: 'string', enum: ['buildings', 'trees'] },
      filters: { type: 'array', items: { type: 'object', required: ['field','op','value'],
        properties: { field: {type:'string'}, op: {type:'string', enum:['eq','neq','gt','lt','between','contains']}, value: {} }}},
      match: { type: 'string', enum: ['all', 'any'] } }}}}},

  { toolSpec: { name: 'features_within', description: 'Find features within a distance of an anchor location. Use for proximity queries.',
    inputSchema: { json: { type: 'object', required: ['anchor_name','distance_m'], properties: {
      anchor_name: { type: 'string' },
      distance_m: { type: 'number', description: 'Buffer radius in metres' },
      target_layer: { type: 'string', enum: ['buildings', 'trees'] },
      mode: { type: 'string', enum: ['within', 'nearest_k'] },
      k: { type: 'number' } }}}}},

  { toolSpec: { name: 'aggregate_features', description: 'Count, sum, or average a numeric field. Optionally group by a dimension.',
    inputSchema: { json: { type: 'object', required: ['metric','layer'], properties: {
      metric: { type: 'string', enum: ['count','sum','avg','min','max'] },
      field: { type: 'string' },
      group_by: { type: 'string' },
      layer: { type: 'string', enum: ['buildings', 'trees'] } }}}}},

  { toolSpec: { name: 'rank_features', description: 'Return top-N features sorted by a field (tallest, oldest, largest, highest FCI).',
    inputSchema: { json: { type: 'object', required: ['sort_field','layer'], properties: {
      sort_field: { type: 'string' },
      direction: { type: 'string', enum: ['desc','asc'] },
      limit: { type: 'number' },
      layer: { type: 'string', enum: ['buildings','trees'] } }}}}}];

// ── System prompt ─────────────────────────────────────────────────
const SYSTEM = `You are CampusGeo, a spatial intelligence assistant for the University of Chicago campus.
You have access to two layers:
- buildings: 308 footprints. Key fields: DISCRIPT1 (name), BD_ID, Year_Completed, BLDG_HGT,
  Gross_Area__s_f__, PROPERTY_S (Owned/Leased), FCI__ (0=good, >0.30=poor), Elevator, Architects, ADDRESS.
- trees: Campus tree inventory. Fields: TreeID, CommonName (format "Type-Variety", e.g. "Linden-Littleleaf"),
  Condition. NOTE: year_planted field not yet populated — do not filter by it.
Always call a tool. Respond in the same language as the user (English or Chinese).
Never invent data. If a query is out of scope, call filter_features with an empty filters array
and set layer to "buildings" — the answer field will explain the limitation.`;

// ── Main handler ──────────────────────────────────────────────────
export const handler = async (event: any) => {
  const { q = '' } = JSON.parse(event.body || '{}');
  if (!q.trim()) return respond(400, { error: 'q is required' });

  // Load layers (parallel, cached in /tmp on warm invocations)
  const [buildings, trees] = await Promise.all([
    loadLayer('layers/buildings.geojson'),
    loadLayer('layers/trees.geojson'),
  ]);

  // Ask Bedrock to classify intent and extract parameters
  const converse = await bedrock.send(new ConverseCommand({
    modelId: MODEL,
    system: [{ text: SYSTEM }],
    messages: [{ role: 'user', content: [{ text: q }] }],
    toolConfig: { tools: TOOLS, toolChoice: { auto: {} } },
  }));

  const toolUse = converse.output?.message?.content?.find((c: any) => c.toolUse)?.toolUse;
  if (!toolUse) return respond(200, { answer: 'I could not understand that query.', features: null, mapAction: null });

  const result = await executeIntent(toolUse.name, toolUse.input, buildings, trees);
  return respond(200, { ...result, intent: toolUse.name });
};

// ── Intent executors ──────────────────────────────────────────────
async function executeIntent(tool: string, input: any, buildings: any, trees: any) {
  const layer = () => input.layer === 'trees' ? trees : buildings;
  const nameOf = (f: any) => (f.properties.DISCRIPT1 || f.properties.CommonName || '').trim();
  const findByName = (feats: any[], q: string) => feats.find(f =>
    [f.properties.DISCRIPT1, f.properties.CommonName, f.properties.OtherNames, f.properties.ADDRESS]
    .some(v => v && v.toLowerCase().includes(q.toLowerCase())));

  switch (tool) {
    case 'locate_feature': {
      const hit = findByName(layer().features, input.name);
      if (!hit) return { answer: `No feature named "${input.name}" found.`, features: null, mapAction: null };
      return { answer: `${nameOf(hit)} is at ${hit.properties.ADDRESS || 'campus'}.`,
        features: { type: 'FeatureCollection', features: [hit] },
        mapAction: { type: 'locate', center: [hit.properties.Lon, hit.properties.Lat], zoom: 17 } };
    }
    case 'filter_features': {
      const hits = layer().features.filter((f: any) => {
        const results = (input.filters || []).map((flt: any) => {
          const v = f.properties[flt.field];
          switch (flt.op) {
            case 'eq': return String(v).toLowerCase() === String(flt.value).toLowerCase();
            case 'neq': return String(v).toLowerCase() !== String(flt.value).toLowerCase();
            case 'gt': return +v > +flt.value;
            case 'lt': return +v < +flt.value;
            case 'contains': return String(v||'').toLowerCase().includes(String(flt.value).toLowerCase());
            case 'between': return +v >= flt.value[0] && +v <= flt.value[1];
            default: return true;
          }
        });
        return input.match === 'any' ? results.some(Boolean) : results.every(Boolean);
      });
      return { answer: `Found ${hits.length} ${input.layer} matching your criteria.`,
        features: { type: 'FeatureCollection', features: hits },
        mapAction: { type: 'highlight' } };
    }
    case 'features_within': {
      const anchor = findByName(buildings.features, input.anchor_name);
      if (!anchor) return { answer: `Could not find "${input.anchor_name}".`, features: null, mapAction: null };
      const pt = turf.centroid(anchor);
      const buf = turf.buffer(pt, input.distance_m / 1000, { units: 'kilometers' });
      const tgt = input.target_layer === 'trees' ? trees : buildings;
      const hits = tgt.features.filter((f: any) => turf.booleanPointInPolygon(turf.centroid(f), buf));
      return { answer: `${hits.length} ${input.target_layer || 'features'} within ${input.distance_m}m of ${nameOf(anchor)}.`,
        features: { type: 'FeatureCollection', features: hits },
        mapAction: { type: 'buffer', center: pt.geometry.coordinates, radiusM: input.distance_m } };
    }
    case 'aggregate_features': {
      const feats = layer().features;
      let answer = '';
      if (input.metric === 'count' && !input.field) {
        if (input.group_by) {
          const g: Record<string,number> = {};
          feats.forEach((f: any) => { const k = String(f.properties[input.group_by]||'Unknown').trim(); g[k]=(g[k]||0)+1; });
          answer = Object.entries(g).sort((a,b)=>b[1]-a[1]).map(([k,v])=>`${k}: ${v}`).join(', ');
        } else { answer = `There are ${feats.length} ${input.layer} on campus.`; }
      } else if (input.field) {
        const vals = feats.map((f:any)=>+f.properties[input.field]).filter(n=>!isNaN(n));
        const r = input.metric==='sum'?vals.reduce((a,b)=>a+b,0):pinput.metric==='avg'?vals.reduce((a,b)=>a+b,0)/vals.length:pinput.metric==='min'?Math.min(...vals):Math.max(...vals);
        answer = `${input.metric} of ${input.field}: ${r.toFixed(2)}.`;
      }
      return { answer, features: null, mapAction: null, meta: { metric: input.metric } };
    }
    case 'rank_features': {
      const sorted = [...layer().features]
        .filter((f:any) => f.properties[input.sort_field] != null)
        .sort((a:any,b:any) => input.direction==='asc' ? a.properties[input.sort_field]-b.properties[input.sort_field] : b.properties[input.sort_field]-a.properties[input.sort_field]);
      const hits = sorted.slice(0, input.limit || 5);
      return { answer: `Top ${hits.length} ${input.layer} by ${input.sort_field}: ${nameOf(hits[0])} is #1.`,
        features: { type: 'FeatureCollection', features: hits },
        mapAction: { type: 'highlight' } };
    }
    default: return { answer: 'Intent not supported yet.', features: null, mapAction: null };
  }
}

const respond = (status: number, body: any) => ({
  statusCode: status,
  headers: { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*' },
  body: JSON.stringify(body),
});
